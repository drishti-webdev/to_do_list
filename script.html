let form = document.getElementById("taskForm"); let taskList = document.getElementById("taskList"); form.addEventListener("submit", function(e){ e.preventDefault(); let title = document.getElementById("title").value; let description = document.getElementById("description").value; addTask(title, description); form.reset(); }); function addTask(title, description){ let div = document.createElement("div"); div.className = "task"; div.innerHTML = ` ${title} ${description} `; div.querySelector(".delete-btn").onclick = function(){ div.remove(); }; taskList.appendChild(div); }
